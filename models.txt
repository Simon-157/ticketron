
class User {
  id: number;
  name: string;
  email: string;
  profilePictureUrl: string;
  likes: number;
  tickets: number;
  following: number;
  socialLinks: SocialLink[];
  favorites: Favorite[];
}

class SocialLink {
  type: string;  // e.g., 'Twitter', 'Instagram', 'Website'
  url: string;
}

class Event {
  id: number;
  title: string;
  date: Date;
  time: string;
  location: string;
  price: Price;
  description: string;
  organizer: Organizer;
  agenda: AgendaItem[];
  images: Image[];
  videoUrl: string;
  ticketsLeft: number;
  category: string; // new field
}

class Price {
  premiumPrice: number;
  regularPrice: number;
}

class Organizer {
  name: string;
  verified: boolean;
}

class AgendaItem {
  title: string;
  speaker: string;
  startTime: string;
  endTime: string;
}

class Image {
  url: string;
}

class Ticket {
  id: number;
  eventId: number;
  userId: number;
  date: Date;
  time: string;
  location: string;
  seat: string;
  barcode: string;
}

class Order {
  id: number;
  userId: number;
  eventId: number;
  ticketType: string;  // 'Premium' or 'Regular'
  price: number;
  fees: number;
  total: number;
  paymentMethod: string;
}

class ContactInfo {
  fullName: string;
  email: string;
  phone: string;
  receiveUpdates: boolean;
}

class Favorite {
  userId: number;
  eventId: number;
}

class SignIn {
  email: string;
  password: string;
}

class SignUp {
  name: string;
  email: string;
  password: string;
  confirmPassword: string;
}

class QRCode {
  eventId: number;
  userId: number;
  qrCodeUrl: string;
}

class ExploreEvent {
  id: number;
  title: string;
  date: Date;
  location: string;
  price: number;
  imageUrl: string;
}
